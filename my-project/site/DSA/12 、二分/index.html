    #include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 2e5 + 10;
ll n, m;
ll a[N], b[N];
bool check(ll mid) {
    ll cnt = m;
    for (int i = 1; i <= n; i++) {
        if (a[i] >= mid)
            continue;
        else {
            ll k = mid - a[i];
            if (k > b[i])
                return false;
            cnt -= k;
            if (cnt < 0)
                return false;
        }
    }
    return true;
}

        cin >> a[i];

        cin >> b[i];

    while (l <= r) {
        mid = l + r >> 1;
        if (check(mid)) {
            ans = mid;
            l = mid + 1;
        } else
            r = mid - 1;
    }
    cout << ans << endl;
    return 0;
}

    return p1.first < p2.first;

    vector<pii> v;
    for (int i = 1; i <= n; i++) {
        if (p[i].second <= mid) {
            ll l = max(1ll, p[i].first - (mid - p[i].second));
            ll r = min(len, p[i].first + (mid - p[i].second));
            v.push_back({l, r});
        }
    }
    sort(v.begin(), v.end(), cmp);
    ll l = v[0].first, r = v[0].second;
    if (l != 1)
        return false;
    for (int i = 1; i < v.size(); i++) {
        ll x = v[i].first, y = v[i].second;
        if (x > r + 1)
            return false;
        else
            r = max(r, y);
    }
    if (r != len)
        return false;
    return true;
}
