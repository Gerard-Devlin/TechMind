void cold_beans() {
    int a, b, c, m;

    vec<int> la(m + 1), ra(m + 1), lb(m + 1), rb(m + 1), lc(m + 1), rc(m + 1), h(m + 1);

        for (int j = 1; j <= b; j++)

            cf[ra[i] + 1][rb[i] + 1][lc[i]] -= h[i];
            cf[ra[i] + 1][rb[i] + 1][rc[i] + 1] += h[i];
