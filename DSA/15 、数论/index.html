    #include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int n, k, ans, a[30];
bool check(int x) {
    if (x <= 1)
        return false;
    for (int i = 2; i <= sqrt(x); i++) {
        if (x % i == 0)
            return false;
    }
    return true;
}
void dfs(int u, int s, int sum) {
    if (s == k) {
        if (check(sum))
            ans++;
        return;
    }
    for (int i = u; i <= n; i++)
        dfs(i + 1, s + 1, sum + a[i]);
}
int main() {
    cin >> n >> k;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    dfs(1, 0, 0);
    cout << ans << endl;
    return 0;
}
